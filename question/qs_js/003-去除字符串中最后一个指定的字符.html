# 003-去除字符串中最后一个指定的字符

知识点：

  • 字符串转数组
  • 从数组最后一个元素开始遍历，第一个出现和指定字符相同的元素，即是要找的元素
function lastWord (str, word) {
    var strArr = [...str],
        strLen = strArr.length
    
    for (var i = strLen - 1; i > -1; i--) {
        if (strArr[i] === word) {
            strArr.splice(i, 1)
            return strArr.join('')
        }
    }
}
console.log('指定字符', lastWord('hstshrstafghj', 'h'))
